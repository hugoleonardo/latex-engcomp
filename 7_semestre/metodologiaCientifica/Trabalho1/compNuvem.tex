\section{Introdução}
\begin{frame}{O que é computação em nuvem?}
\begin{itemize}
	\item Por que usar o termo "nuvem"?
	\begin{itemize}
		\item Engloba um ambiente de interior desconhecido
		\item Só se conhece seu início e seu fim
		\item Esconde uma série de serviços e aplicações
	\end{itemize}
	\item Novo modelo de computação
	\item Acesso à aplicações e serviços independentemente de plataforma
	\item Requer conexão com a internet
	\item \textit{Datacenters} para processamento e armazenamento de dados
\end{itemize}
\end{frame}

\section{Camadas de arquitetura e tipos de nuvem}
\begin{frame}{Classes de serviço}
\begin{itemize}
	\item Dividido nas camadas:
	\begin{itemize}
		\item Inferior
		\item Intermediária
		\item Superior
	\end{itemize}
	\item Dividido nos tipos:
	\begin{itemize}
		\item Infra-estrutura como serviço - IaaS(Infrastructure as a Service)
		\item Plataforma como serviço - PaaS(Plataform as a Service)
		\item Software como serviço - SaaS(Software as a Service)
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Infra-estrutura como Serviço - IaaS}
\begin{itemize}
	\item Recursos de hardware virtualizados
	\begin{itemize}
		\item Computação
		\item Armazenamento
		\item Comunicação
	\end{itemize}
	\item Servidores para softwares customizadas e sistemas operacionais
	\item Interface única de administração da infra-estrutura
	\begin{itemize}
		\item Comunicação entre hosts, switches, roteadores e inclusão de equipamentos novos
	\end{itemize}
	\item Pertence à camada inferior e dá suporte para as outras camadas
\end{itemize}
\end{frame}

\begin{frame}{Plataforma como Serviço - PaaS}
\begin{itemize}
	\item Criação e implementação de aplicações pelo desenvolvedor
	\item Quantidade de processadores e memória necessários para a execuçao não são responsabilidade do desenvolvedor
	\item Alto nível de integração e compatibilidade
	\begin{itemize}
		\item Sistemas Operacionais
		\item Linguagem de programação
		\item Ambientes de desenvolvimento
	\end{itemize}
\end{itemize}
\end{frame}
