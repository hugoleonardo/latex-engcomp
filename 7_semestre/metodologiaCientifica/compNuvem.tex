\section{Introdução}
\begin{frame}{O que é computação em nuvem?}
\begin{itemize}
	\item Por que usar o termo "nuvem"?
	\begin{itemize}
		\item Engloba um ambiente de interior desconhecido
		\item Só se conhece seu início e seu fim
		\item Esconde uma série de serviços e aplicações
	\end{itemize}
	\item Novo modelo de computação
	\item Acesso à aplicações e serviços independentemente de plataforma
	\item Geralmente requer conexão com a internet, mas em casos de nuvens locais não é necessário
	\item \textit{Datacenters} para processamento e armazenamento de dados
\end{itemize}
\end{frame}

\section{Camadas de arquitetura e tipos de nuvem}
\begin{frame}{Classes de serviço}
\begin{itemize}
	\item Dividido nas camadas:
	\begin{itemize}
		\item Inferior
		\item Intermediária
		\item Superior
	\end{itemize}
	\item Dividido nos tipos:
	\begin{itemize}
		\item Infra-estrutura como serviço - IaaS(Infrastructure as a Service)
		\item Plataforma como serviço - PaaS(Plataform as a Service)
		\item Software como serviço - SaaS(Software as a Service)
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Infra-estrutura como Serviço - IaaS}
\begin{itemize}
	\item Recursos de hardware virtualizados
	\begin{itemize}
		\item Computação
		\item Armazenamento
		\item Comunicação
	\end{itemize}
	\item Servidores para softwares customizadas e sistemas operacionais
	\item Interface única de administração da infra-estrutura
	\begin{itemize}
		\item Comunicação entre hosts, switches, roteadores e inclusão de equipamentos novos
	\end{itemize}
	\item Pertence à camada inferior e dá suporte para as outras camadas
\end{itemize}
\end{frame}

\begin{frame}{Exemplo - IaaS}
\begin{itemize}
	\item Amazon EC2
	\begin{itemize}
		\item AMI(Amazon Machine Image) pré-configurada 
		\item AMI customizada
		\item Segurança e acesso à rede
		\item Iniciar, parar e monitorar instâncias de AMI
		\item API de serviço web ou outras ferramentas de gerência
		\item Rodar em vários locais, utilizar IP estático, adicionar blocos de armazenamento
		\item Tarifa: horas de execução ou transferência de dados
		\item 750 grátis por mês durante 1 ano
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Plataforma como Serviço - PaaS}
\begin{itemize}
	\item Criação e implementação de aplicações pelo desenvolvedor
	\item Quantidade de processadores e memória necessários para a execuçao não são responsabilidade do desenvolvedor
	\item Alto nível de integração e compatibilidade
	\begin{itemize}
		\item Sistemas Operacionais
		\item Linguagem de programação
		\item Ambientes de desenvolvimento
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Software como Serviço - SaaS}
\begin{itemize}

	\item O usuário não tem controle de como os processos ocorrem, das configurações de hardware e nem possui o poder de criar novos programas  
	\item É disponibilizado um software final para utilização por parte do usuário
	\begin{itemize}
		\item GoogleDocs
		\item QuakeLive
		\item FreePDF Converter

\section{Modelos de Implementação}
\begin{frame}{Modelos de Implementação}
\begin{itemize}
	\item O modelo de implementação está relacionado com quem poderá utilizar a nuvem por completo
	\item Existem 4 modos de implementação 
	\begin{itemize}
		\item Público
		\item Privado
		\item Comunidade
		\item Hibrído
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Modelos de Implementação}
\begin{itemize}
	\item Modelo Privado
	\begin{itemize}
		\item Nuvem local com utilização restrita
		\item A infra-estrutura e o gerencioamento da nuvem ficará nas mãos do próprio dono como por exemplo uma empresa. 
		\item A manutenção poderá ser feita pela própria empresa ou por terceiros
		\item Alto grau de segurança
	\end{itemize}
	\item Modelo Público
	\begin{itemize}
		\item Utilização aberta para os usuarios(internet)
		\item Não significa ser um serviço gratuito
		\item Baixo grau de segurança
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Modelos de Implementação}
\begin{itemize}
	\item Modelo Comunidade
	\begin{itemize}
		\item Utilização é dividida entre várias empresas gerando assim uma comunidade
		\item A infra-estrutura e o gerencioamento da nuvem será dividida pelas empresas da comunidade 
	\end{itemize}
	\item Modelo Híbrido
	\begin{itemize}
		\item Utiliza nuvens privadas e públicas para conseguir seu objetivo
		\item Processos básicos podem ser feitos em nuvens públicas em vez de utilizar os recursos da nuvem privada
	\end{itemize}
\end{itemize}
\end{frame}

\section{Conclusões}
\begin{frame}{Conclusões}
\begin{itemize}
	\item Vantagens
	\begin{itemize}
		\item Geralmente não existe preocupação com as características do computador local, pois a nuvem ja possui o sistema operacional e os programas necessários para o usuário
		\item A manutenção da infra-estrutura, atualização de softwares, armazenamento e leitura de dados são todos processos transparentes ao usuário
		\item melhor controle nos gastos com aplicativos, pois geralmente se paga pelo tempo de uso e não por uma licensa integral
		\item pouca necessidade de espaço físico, pois a nuvem ja contém todos os programas e armazena os dados necessários para o usuário final
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Conclusões}
\begin{itemize}
\item Desvantagens
	\begin{itemize}
		\item Em alguns casos necessita de uma conexão com a internet rápida
		\item Comprometimento da privacidade e segurança dos arquivos, pois estes estão na nuvem e não no computador pessoal do usuário
	\end{itemize}
\end{itemize}
\end{frame}
