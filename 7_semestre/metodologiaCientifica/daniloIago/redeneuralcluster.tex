\section{Introdução}

\subsection{Introdução Geral}

		\begin{frame}{Introdução do Artigo}
		\begin{itemize}
			\item Alto consumo de energia em \textit{datacenters} (40\% para equipamentos e 60\% para infra-estrutura)
			\item Virtualizar servidores reduz o consumo de energia
				\begin{itemize}
					\item Gera maior ociosidade 			
				\end{itemize}
			\item Normalmente os equipamentos são superdimensionados (ociosidade)
			\item O autor propõe uma nova política de gerência de servidores \textit{web}
		\end{itemize}		
	\end{frame}

\subsection{Introdução à Virtualização}

	\begin{frame}{Arquitetura do Xen}
		\begin{itemize}

			\item Possui um \textit{kernel} modificado (\textit{dom0}
			\item O \textit{hypervisor} faz o mapeamento de recursos físicos em virtuais
			\item As VM's rodam no Domínio U (\textit{domU}
		\end{itemize}

\subsection{Introdução à Redes Neurais Artificias (RNA)}

	\begin{frame}{O que é uma RNA?}
		\begin{itemize}
			\item São modelos computacionais
				\begin{itemize}
					\item Adquirir, armazenar e utilizar conhecimento
					\item Generalizar determinados sistemas
					\item Agrupar e organizar dados
				\end{itemize}
			\item Compostos por neurônios (unidades de processamento)
			\item Interligada por sinapses (conexões)
			\item Agrupados em camadas
			\begin{figure}[h]
				\centering
				\includegraphics[scale=0.35]{./pictures/redes_neurais_camadas.png}
				\label{fig:redesneuraiscamadas}
			\end{figure}			
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Funcionamento da RNA}
		\begin{itemize}
			\item Neurônios são utilizados para armazenar informações
			\item Cada sinapse é caracterizada por um peso
			\item O somador adiciona as entradas ponderadas
			\item A função de ativação restringe a saída entre [0,1] ou [-1,1]
			\item O \textit{bias} é utilizado para aumentar ou diminuir a entrada da função de ativação
		\end{itemize}
			\begin{figure}[h]
				\centering
				\includegraphics[scale=0.45]{./pictures/neuronio_biologico_artificial.png}
				\label{fig:neuroniobiologicoartificial}
			\end{figure}	
	\end{frame}
\section{Trabalhos Relacionados}

\subsection{Gerência de VM's}

	\begin{frame}{Gerência de VM's}
		O autor cita três trabalhos sobre gerência inteligente de VM's
		\begin{itemize}
			\item Utilizando Balanceamento de carga
				\begin{itemize}
					\item Analisa os recursos dos \textit{hosts} [Megav 2007]
					\item Migra as VM's de acordo com a quantdade de recursos disponíveis
				\end{itemize}
			\item Utilizando um controlador \textit{lookahead} [Kusic et al. 2008]
				\begin{itemize}
					\item Analisa a quantidade de conexões e o tempo médio de resposta
					\item Define a quantidade de \textit{hosts} que devem ser ativados
				\end{itemize}
			\item Utilizando múltiplos domínios [Ruth et al. 2006]
				\begin{itemize}
					\item Um \textit{host} e suas VM's formam um domínio
					\item O sistema analisa	a demanda de CPU e memória das VM's
					\item O sistema migra as VM's caso os recursos de um domínio estejam esgotados
				\end{itemize}
		\end{itemize}
	\end{frame}

\subsection{Balanceamento de Carga utilizando RNA}

	\begin{frame}{Balanceamento de Carga utilizando RNA}
		O autor cita três trabalhos referentes ao uso de RNA's para balanceamento de carga	
		\begin{itemize}
			\item Controlar frequências de operação de CPU's [Souza et al. 2008] 
				\begin{itemize}
					\item Utiliza DFVS (\textit{Dynamic Voltage ans Frequency Scaling} para controlar o 								\textit{clock} das CPU's
					\item Prove economia de energia mantendo o nível do serviço
					\item Foram mapeadas frequência e taxa de ocupação das CPU's para associação com consumo 						de energia
				\end{itemize}
			\item Classificar estados de um servidor \textit{web} utilizando RNA [Simula et al. 2008]
				\begin{itemize}		
					\item Mapas auto-organizáveis com nove dimensões de entrada
					\item Foram medidos uso de CPU, pacotes recebidos, blocos lidos/escritos do disco, etc.
					\item A rede apresentou 4 estados de ocupação do servidor
					\item Criação de uma política de tolerância a falhas baseada nesses estados
				\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Balanceamento de Carga utilizando RNA}
		\begin{itemize} 		
 			\item Classificar uso de memoria utilizando RNA [Lin 2006]
				\begin{itemize}
					\item Utiliza 3 indicadores de consumo de memória como entrada
					\item A rede indicou 6 estados de consumo de memória
					\item Um módulo atuador foi projeto com base nos valores desses estados
					\item Este módulo substitui o \textit{OOM Killer} nativo do \textit{Linux}
				\end{itemize}
		\end{itemize}	
	\end{frame}

\section{Arquitetura Proposta}


\subsection{Sistema proposto}
	\begin{frame}{Abordagem}
		\begin{itemize}
			\item Substituir a política de gerenciamento original por rede neural, aprimorando os mecanismos de atuação no cluster
			\item Definir os estados de operação no cluster de acordo com os parâmetros analisados
			\item Propor intervenções específicas da API de atuação de acordo com o estado do cluster
			\item Objetivo: economia de energia e manutenção do tempo de resposta das requisições
		\end{itemize}
	\end{frame}


	\begin{frame}{Sistema Proposto}
		\begin{figure}
			\centering
			\includegraphics[scale=0.45]{./pictures/arquitetura.png}
		\end{figure}
	\end{frame}
	

\begin{frame}{Analisando Sistema Proposto}
	\begin{itemize}
		\item Controlador: fornece a classificação do estado corrente conforme as medições coletadas
		\item Atuador: aciona as funções de gerência da API de acordo com a ordem do Controlador 
		\item Coletor: obtém através da API os dados monitorados por agentes que atuam nos hospedeiros do cluster
		\item Balanceador: recebe requisições do usuário e as distribui para as VMs executadas nos servidores físicos hospedeiros. Deve reconhecer o estado atual e as diferentes localizações das VMs
	\end{itemize}
\end{frame}

	\begin{frame}{Políticas de Reconfiguração}
