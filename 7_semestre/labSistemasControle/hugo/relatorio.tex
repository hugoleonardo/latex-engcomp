\section{Questao 1}

	Simulação de um sistema com entrada igual a um degrau e com função de transferência igual à equação \ref{eq:sistordem1}. Na primeira parte, manteve-se o K constante e variou-se o $\tau$ entre os valores 0.1, 1 e 10. No segundo caso, $\tau$ foi mantido constante e variou-se o K entre os valores 0.1, 1 e 10. O critério de avaliaçao é o tempo de estabilização $T_{s}$).
	
\begin{equation}\label{eq:sistordem1}
	\frac{C(s)}{R(s)} = \frac{K}{\tau s+1}
\end{equation}

\subsection{Simulação}

	No item \textit{i} percebeu-se que o $T_{s}$ subia praticamente com a mesma proporção de crescimento do K. Como mostrado nos dados logo abaixo e nas Figuras \ref{fig:questao1k2tau01}, \ref{fig:questao1k2tau1} e \ref{fig:questao1k2tau01}
\begin{itemize}
	\item Item i
	\begin{itemize} 
		\item Para $K = 2$ e $\tau = 0,1$ : $T_{s} = 0,37$ mostrado na Figura \ref{fig:questao1k2tau01}
		
		\item Para $K = 2$ e $\tau = 1$ : $T_{s} = 3,957$ mostrado na Figura \ref{fig:questao1k2tau1}

		\item Para $K = 2$ e $\tau = 10$ : $T_{s} = 39,18$ mostrado na Figura \ref{fig:questao1k2tau10}
		
	\end{itemize}
\end{itemize}

	No item \textit{ii} percebeu-se que o $T_{s}$ permaneceu constante devido a independência de K com o valor do $T_{s}$. Como mostrado nos dados logo abaixo e nas Figuras \ref{fig:questao1k01tau1}, \ref{fig:questao1k1tau1} e \ref{fig:questao1k10tau1}.
\begin{itemize}
	\item Item ii
	\begin{itemize}
		\item Para $\tau=1$ e $K=0,1$ : $T_{s} = 3,957$ (Figura \ref{fig:questao1k01tau1})
		
		\item Para $\tau=1$ e $K=1$ : $T_{s} = 3,957$ (Figura \ref{fig:questao1k1tau1})
		
		\item Para $\tau=1$ e $K=10$ : $T_{s} = 3,957$ (Figura \ref{fig:questao1k10tau1})
		
	\end{itemize}	
\end{itemize}

	O Cálculo da resposta analítica (no domínio de tempo) é dado abaixo, utilizou-se expansão em frações parciais e transformada de Laplace.
			
		\begin{itemize}
			\item Item i
				\begin{itemize}
					\item $K=2$, $\tau = 0,1$	
						\[ \frac{1}{s}\frac{2}{0,1s+1}\]
						\[ \frac{A}{s}\frac{B}{0,1s+1}\]
						\[ A = s\frac{2}{s(0,1s+1)}\mid s=0 = 2\]
						\[ B = 0,1s+1\frac{2}{s(0,1s+1)}\mid s=-10 = -0,2\]
						\[ C(s) = \frac{2}{s} - \frac{0,1}{s+10}\]
						\[ c(t) = 2 - 0,1e^{-10t}\]
					\item $K=2$, $\tau = 1$
						\[ \frac{1}{s}\frac{2}{1s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{2}{s(s+1)}\mid s=0 = 2\]
						\[ B = s+1\frac{2}{s(s+1)}\mid s=-1 = -2\]
						\[ c(t) = 2 - 2e^{-t}\]
					\item $K=2$, $\tau = 10$
						\[ \frac{1}{s}\frac{2}{10s+1}\]
						\[ \frac{A}{s}\frac{B}{10s+1}\]
						\[ A = s\frac{2}{s(10s+1)}\mid s=0 = 2\]
						\[ B = 10s+1\frac{2}{s(10s+1)}\mid s=-1/10 = -20\]
						\[ c(t) = 2 - 2e^{-t/10}\]
					
							\end{itemize}	
			\item Item ii
				\begin{itemize}
					\item $\tau=1$, $K=0,1$
						\[ \frac{1}{s}\frac{0,1}{s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{0,1}{s(s+1)}\mid s=0 = 0,1\]
						\[ B = s+1\frac{0,1}{s(s+1)}\mid s=-1 = -0,1\]
						\[ c(t) = 0,1 - 0,1e^{-t}\]
					\item $\tau=1$, $K=1$ 			
						\[ \frac{1}{s}\frac{1}{s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{1}{s(s+1)}\mid s=0 = 1\]
						\[ B = s+1\frac{1}{s(s+1)}\mid s=-1 = -1\]
						\[ c(t) = 1 - 1e^{-t}\]
					\item $\tau=1$, $K=10$ 			
						\[ \frac{10}{s}\frac{1}{s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{10}{s(s+1)}\mid s=0 = 10\]
						\[ B = s+1\frac{10}{s(s+1)}\mid s=-1 = -10\]
						\[ c(t) = 10 - 10e^{-t}\]
				\end{itemize}
		\end{itemize}	

\section{Questão 2}

	Simulação de um sistema de $2^{a}$ ordem com função de transferência igual à equação \ref{eq:sistordem2}. Na primeira parte, manteve-se $w_{n}$ em 2 e K em 8 constantes e variou-se o $\xi$ entre 1, 0.7 e 0.2. No segundo caso, foram usados os mesmos cirtérios da primeira parte, porém $w_{n}$ valia 10. \begin{equation}\label{eq:sistordem2}
	\frac{C(s)}{R(s)} = \frac{Kw_{n}^{2}}{s^{2} + 2\xi w_{n}s + w_{n}^{2}}
