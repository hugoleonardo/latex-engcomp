		\begin{itemize}
			\item Item i
				\begin{enumerate}
					\item $K=2$, $\tau = 0,1$	
						\[ \frac{1}{s}\frac{2}{0,1s+1}\]
						\[ \frac{A}{s}\frac{B}{0,1s+1}\]
						\[ A = s\frac{2}{s(0,1s+1)}\mid s=0 = 2\]
						\[ B = 0,1s+1\frac{2}{s(0,1s+1)}\mid s=-10 = -0,2\]
						\[ C(s) = \frac{2}{s} - \frac{0,1}{s+10}\]
						\[ c(t) = 2 - 0,1e^{-10t}\]
					\item $K=2$, $\tau = 1$
						\[ \frac{1}{s}\frac{2}{1s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{2}{s(s+1)}\mid s=0 = 2\]
						\[ B = s+1\frac{2}{s(s+1)}\mid s=-1 = -2\]
						\[ c(t) = 2 - 2e^{-t}\]
					\item $K=2$, $\tau = 10$
						\[ \frac{1}{s}\frac{2}{10s+1}\]
						\[ \frac{A}{s}\frac{B}{10s+1}\]
						\[ A = s\frac{2}{s(10s+1)}\mid s=0 = 2\]
						\[ B = 10s+1\frac{2}{s(10s+1)}\mid s=-1/10 = -20\]
						\[ c(t) = 2 - 2e^{-t/10}\]
					
							\end{enumerate}	
			\item Item ii
				\begin{enumerate}
					\item $\tau=1$, $K=0,1$
						\[ \frac{1}{s}\frac{0,1}{s+1}\]
						\[ \frac{A}{s}\frac{B}{s+1}\]
						\[ A = s\frac{0,1}{s(s+1)}\mid s=0 = 0,1\]
						\[ B = s+1\frac{0,1}{s(s+1)}\mid s=-1 = -0,1\]
						\[ c(t) = 0,1 - 0,1e^{-t}\]
					\item $\tau=1$, $K=1$ 			
						\[ \frac{1}{s}\frac{1}{s+1}\]

		\begin{equation}\label{calcwd}
			w_{d} = w_{n}\sqrt{1-\xi^{2}}
		\end{equation}
		
		\begin{equation}\label{calctp}
			T_{p} = \frac{\pi}{w_{d}}
		\end{equation}

					\item $w_{n} = 2$ e $\xi = 0,7$

						\[ T_{p} = \frac{\pi}{1,428} = 2,19\]
					\item $w_{n} = 2$ e $\xi = 0,2$
						\[ w_{d} = 2\sqrt{1-0,2^{2}} = 1,9595\]
						\[ T_{p} = \frac{\pi}{1,9595} = 1,603\]
				\end{enumerate}
			\item Item ii
				\begin{enumerate}
					\item $w_{n} = 10$ e $\xi = 0,7$
						\[ w_{d} = 10\sqrt{1-0,7^{2}} = 7,1414\]
						\[ T_{p} = \frac{\pi}{7,1414} = 0,4399\]
					\item $w_{n} = 10$ e $\xi = 0,2$
						\[ w_{d} = 10\sqrt{1-0,2^{2}} = 9,7979\]
						\[ T_{p} = \frac{\pi}{9,7979} = 0,32\]
				
				\end{enumerate}
		\end{itemize}
